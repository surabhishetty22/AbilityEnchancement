
#include <iostream>
using namespace std;
int main() 
{
    int i,n,a[100];
    cout<<"enter no of elements";
    cin>>n;
    cout<<"enter elements";
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }
    cout<<"order is\n";
    for(i=0;i<n;i++)
    {
    cout<<a[i]<<" ";
    }
    cout<<"\n reversed order is";
    for(i=(n-1); i>=0; i--)
        cout<<a[i]<<" ";
    
}
